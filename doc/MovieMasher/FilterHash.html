def filter_command scope = nil, job_output = nil
  cmd = super
  unless cmd.empty? then # I'm not disabled
    cmds = Array.new
    @hash.each do |name, value|
      name = name.id2name if name.is_a? Symbol
      cmds << "#{name}=#{value}"
    end
    cmd = "#{cmd}=#{cmds.join ':'}" unless cmds.empty?
  end
  cmd
end
