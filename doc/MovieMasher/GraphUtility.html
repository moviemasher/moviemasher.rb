def __join_commands(cmds)
  joined_commands = []
  cmds = cmds.reject(&:empty?)
  c = cmds.length
  c.times do |i|
    cmd = cmds[i]
    cmd = "#{cmd}," unless (i.zero? && cmd.end_with?(':v]')) || i == c - 1
    joined_commands << cmd
  end
  joined_commands.join
end
