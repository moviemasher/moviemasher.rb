def prefixed_configuration(key, hash)
  config = {}
  hash.each do |k, value|
    k_str = k.id2name
    next unless k_str.start_with?(key)
    next if value.to_s.empty?
    k_str[key] = ''
    config[k_str.to_sym] = value
  end
  config
end
