def directory_files(file)
  uploading_directory = File.directory?(file)
  files = []
  if uploading_directory
    file = Path.add_slash_end file
    Dir.entries(file).each do |f|
      f = file + f
      files << f unless File.directory?(f)
    end
  else
    files << file
  end
  files
end
